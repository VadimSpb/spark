    def ratingToClass(rating):
        article_class = 'A'
        if rating > 100: 
            article_class = 'D'
        elif rating > 50:
            article_class = 'C'
        elif rating > 0:
            article_class = 'B'
        elif rating == 0:
            article_class = 'A'
            
        return article_class
